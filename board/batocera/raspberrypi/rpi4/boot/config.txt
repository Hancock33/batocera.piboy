								# CEA 97=2160p@60Hz, 16=1080p@60Hz, 4=720p@60Hz

hdmi_force_hotplug:1=1			# Setting hdmi_force_hotplug to 1 pretends that the HDMI hotplug signal is asserted, so it appears that a HDMI display is attached.
